import java.util.*;
public class TriangleValidityChecker {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n1 = sc.nextInt();
		int n2 = sc.nextInt();
		int n3 = sc.nextInt();
		
		if((n1+n2)>n3 && (n2+n3)>n1 && (n1+n3)>n2)
			System.out.println("Triangle");
		else
			System.out.println("Not a Triangle");
	}

}
